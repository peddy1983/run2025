# 本機開發環境變數範本
# 複製這個檔案的內容到 .env 檔案中
# 檔案位置：專案根目錄（與 package.json 同一層）

# ========================================
# Firebase 前端設定（必填）
# ========================================
# 請到 Firebase Console 取得這些資訊
# 位置：Firebase Console > 專案設定 > 一般 > 您的應用程式 > SDK 設定和設定

VITE_FIREBASE_API_KEY=your-firebase-api-key
VITE_FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=your-project-id
VITE_FIREBASE_STORAGE_BUCKET=your-project.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=123456789
VITE_FIREBASE_APP_ID=1:123456789:web:abcdef

# ========================================
# LINE Login 前端設定（必填）
# ========================================
# 請到 LINE Developers Console 取得這些資訊
# 位置：LINE Developers Console > 您的 Channel > Basic settings

VITE_LINE_CHANNEL_ID=your-line-channel-id
VITE_LINE_CALLBACK_URL=http://localhost:5173/auth/callback

# ========================================
# 使用說明
# ========================================
# 1. 將上方的 your-xxxx 替換成實際的值
# 2. 儲存後重新啟動開發伺服器（npm run dev）
# 3. .env 檔案不會被提交到 Git（已在 .gitignore 中）

# ========================================
# 注意：Netlify 部署需要額外設定
# ========================================
# 以下環境變數需要在 Netlify Dashboard 設定（不要放在 .env）：
#
# 前端變數（VITE_ 開頭）：
# - VITE_FIREBASE_API_KEY
# - VITE_FIREBASE_AUTH_DOMAIN
# - VITE_FIREBASE_PROJECT_ID
# - VITE_FIREBASE_STORAGE_BUCKET
# - VITE_FIREBASE_MESSAGING_SENDER_ID
# - VITE_FIREBASE_APP_ID
# - VITE_LINE_CHANNEL_ID
# - VITE_LINE_CALLBACK_URL=https://your-app.netlify.app/auth/callback
#
# 後端變數（Functions 專用）：
# - FIREBASE_PROJECT_ID=your-project-id
# - FIREBASE_CLIENT_EMAIL=firebase-adminsdk-xxxxx@your-project.iam.gserviceaccount.com
# - FIREBASE_PRIVATE_KEY="-----BEGIN PRIVATE KEY-----\n...\n-----END PRIVATE KEY-----\n"
# - LINE_CHANNEL_ID=your-line-channel-id
# - LINE_CHANNEL_SECRET=your-line-channel-secret
#
# 詳細設定步驟請參考 LINE_LOGIN_SETUP.md
